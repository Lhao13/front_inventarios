--------------------actualizar_activo_pc

CREATE OR REPLACE FUNCTION actualizar_activo_pc(
    p_id_activo integer,
    p_numero_serie text,
    p_nombre text,
    p_codigo integer,
    p_id_tipo_activo integer,
    p_id_condicion_activo integer,
    p_id_custodio integer,
    p_id_ciudad_activo integer,
    p_id_sede_activo integer,
    p_id_area_activo integer,
    p_id_provedor integer,
    p_fecha_adquisicion date,
    p_ip text,
    p_fecha_entrega date,
    p_coordenada text,

    p_procesador text,
    p_ram text,
    p_almacenamiento text,
    p_id_marca integer,
    p_modelo text,
    p_cargador_codigo text,
    p_num_puertos integer,
    p_observaciones text
)
RETURNS void
LANGUAGE plpgsql
AS $$
DECLARE
    v_categoria text;
BEGIN

    -- Validar que el activo sea PC
    SELECT categoria_activo
    INTO v_categoria
    FROM activo
    WHERE id = p_id_activo;

    IF v_categoria IS NULL THEN
        RAISE EXCEPTION 'El activo no existe';
    END IF;

    IF v_categoria <> 'PC' THEN
        RAISE EXCEPTION 'El activo % no es de categoria PC', p_id_activo;
    END IF;

    -- Update activo
    UPDATE activo SET
        numero_serie = p_numero_serie,
        nombre = p_nombre,
        codigo = p_codigo,
        id_tipo_activo = p_id_tipo_activo,
        id_condicion_activo = p_id_condicion_activo,
        id_custodio = p_id_custodio,
        id_ciudad_activo = p_id_ciudad_activo,
        id_sede_activo = p_id_sede_activo,
        id_area_activo = p_id_area_activo,
        id_provedor = p_id_provedor,
        fecha_adquisicion = p_fecha_adquisicion,
        ip = p_ip,
        fecha_entrega = p_fecha_entrega,
        coordenada = p_coordenada
    WHERE id = p_id_activo;

    -- Update info_pc
    UPDATE info_pc SET
        procesador = p_procesador,
        ram = p_ram,
        almacenamiento = p_almacenamiento,
        id_marca = p_id_marca,
        modelo = p_modelo,
        cargador_codigo = p_cargador_codigo,
        num_puertos = p_num_puertos,
        observaciones = p_observaciones
    WHERE id_activo = p_id_activo;

END;
$$;

CREATE OR REPLACE FUNCTION eliminar_activo(p_id_activo integer)
RETURNS void
LANGUAGE plpgsql
SECURITY DEFINER
AS $$
BEGIN

    DELETE FROM info_pc WHERE id_activo = p_id_activo;
    DELETE FROM info_equipo_comunicacion WHERE id_activo = p_id_activo;

create or replace function public.fn_historial_activo_insert()
returns trigger
language plpgsql
security definer
set search_path = public
as $$
begin

  -- Asegurar timestamp en activo si viene null
  if new.timestamp_created_at is null then
    new.timestamp_created_at := now();
  end if;

  -- Insertar en historial_activo
  insert into public.historial_activo (
    id_activo,
    tipo_operacion,
    timestamp_changed_at,
    user_on_change,
    snapshot_json
  )
  values (
    new.id,
    'INSERT',
    now(),
    auth.uid(),
    to_jsonb(new)
  );

  return new;
end;
$$;


-- =========================================
-- 2️⃣ CREAR TRIGGER
-- =========================================

drop trigger if exists trg_historial_activo_insert on public.activo;

create trigger trg_historial_activo_insert
after insert
on public.activo
for each row
execute function public.fn_historial_activo_insert();

create or replace function public.fn_validar_categoria_activo()
returns trigger
language plpgsql
as $$
declare
    v_categoria categoria_activo_enum;
    total integer;
begin

    -- Obtener categoria del activo
    select categoria_activo
    into v_categoria
    from public.activo
    where id = new.id_activo;

    if v_categoria is null then
        raise exception 'El activo no tiene categoria definida.';
    end if;

    -- Validar que la tabla coincida con la categoria
    if
        (tg_table_name = 'info_pc' and v_categoria <> 'PC') or
        (tg_table_name = 'info_equipo_comunicacion' and v_categoria <> 'COMUNICACION') or
        (tg_table_name = 'info_software' and v_categoria <> 'SOFTWARE') or
        (tg_table_name = 'info_equipo_generico' and v_categoria <> 'GENERICO')
    then
        raise exception 'La categoria del activo no coincide con la tabla destino.';
    end if;

    -- Verificar que no exista en más de una tabla
    select
        (select count(*) from public.info_pc where id_activo = new.id_activo) +
        (select count(*) from public.info_equipo_comunicacion where id_activo = new.id_activo) +
        (select count(*) from public.info_software where id_activo = new.id_activo) +
        (select count(*) from public.info_equipo_generico where id_activo = new.id_activo)
    into total;

    if total > 1 then
        raise exception 'Un activo solo puede pertenecer a UNA tabla hija.';
    end if;

    return new;
end;
$$;

--info_pc
create trigger trg_validar_pc
